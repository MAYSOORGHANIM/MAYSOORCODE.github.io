مثال HTML: هيكل صفحة ويب أساسية

<!DOCTYPE html>
<html lang="ar" dir="rtl">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>صفحتي الأولى</title>
</head>
<body>
    <header>
        <h1>مرحباً بالعالم!</h1>
    </header>
    <main>
        <p>هذه هي صفحتي الأولى باستخدام HTML.</p>
        <ul>
            <li>عنصر قائمة 1</li>
            <li>عنصر قائمة 2</li>
            <li>عنصر قائمة 3</li>
        </ul>
    </main>
    <footer>
        <p>جميع الحقوق محفوظة © 2023</p>
    </footer>
</body>
</html>

مثال CSS: تنسيق صفحة ويب

/* تنسيق عام للصفحة */
body {
    font-family: Arial, sans-serif;
    line-height: 1.6;
    margin: 0;
    padding: 0;
    background-color: #f4f4f4;
}

/* تنسيق الهيدر */
header {
    background-color: #333;
    color: white;
    padding: 20px;
    text-align: center;
}

/* تنسيق العنوان الرئيسي */
h1 {
    color: #2c3e50;
    margin-bottom: 20px;
}

/* تنسيق الفقرات */
p {
    color: #555;
    margin-bottom: 15px;
}

/* تنسيق القوائم */
ul {
    list-style-type: square;
    margin-right: 20px;
}

li {
    margin-bottom: 10px;
}

/* تنسيق الفوتر */
footer {
    background-color: #333;
    color: white;
    text-align: center;
    padding: 10px;
    position: fixed;
    bottom: 0;
    width: 100%;
}

مثال C++: برنامج حسابي بسيط

#include <iostream>
using namespace std;

// دالة لجمع عددين
int add(int a, int b) {
    return a + b;
}

// دالة لطرح عددين
int subtract(int a, int b) {
    return a - b;
}

// دالة رئيسية
int main() {
    int num1, num2;
    
    cout << "أدخل الرقم الأول: ";
    cin >> num1;
    
    cout << "أدخل الرقم الثاني: ";
    cin >> num2;
    
    // استدعاء الدوال وعرض النتائج
    cout << "مجموع الرقمين: " << add(num1, num2) << endl;
    cout << "فرق الرقمين: " << subtract(num1, num2) << endl;
    
    return 0;
}

مثال Python: آلة حاسبة بسيطة

# دالة لجمع عددين
def add(a, b):
    return a + b

# دالة لطرح عددين
def subtract(a, b):
    return a - b

# دالة لضرب عددين
def multiply(a, b):
    return a * b

# دالة لقسمة عددين
def divide(a, b):
    if b != 0:
        return a / b
    else:
        return "لا يمكن القسمة على صفر!"

# البرنامج الرئيسي
print("آلة حاسبة بسيطة")
print("1. جمع")
print("2. طرح")
print("3. ضرب")
print("4. قسمة")

choice = input("اختر العملية (1/2/3/4): ")

num1 = float(input("أدخل الرقم الأول: "))
num2 = float(input("أدخل الرقم الثاني: "))

if choice == '1':
    print(f"النتيجة: {add(num1, num2)}")
elif choice == '2':
    print(f"النتيجة: {subtract(num1, num2)}")
elif choice == '3':
    print(f"النتيجة: {multiply(num1, num2)}")
elif choice == '4':
    print(f"النتيجة: {divide(num1, num2)}")
else:
    print("اختيار غير صحيح")
